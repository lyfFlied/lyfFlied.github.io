    function ddd1(arr){
    let res = Array.from(new Set(arr));
    return res;
}

    2.filter && indexOf

    function ddd2(arr){
    let res = arr.filter((item,index,arr)=>{
        return arr.indexOf(item) == index;
    });
    return res;
}

    3. forEach && includes

    function ddd4(arr){
    let res = [];
    arr.forEach(item=>{
        if(!res.includes(item)) res.push(item);
    });
    return res;
}

    function ddd4(arr){
    const tem = new Map();
    let res = arr.filter(itme=>{
        return !tem.has(item) && tem.set(item,1);

}
